    type Fmt

    type Fmt struct {
    C  string  // color
    M  string  // marker
    Ls string  // linestyle
    Lw float64 // linewidth; -1 => default
    Ms int     // marker size; -1 => default
    L  string  // label
    Me int     // mark-every; -1 => default
}

    Fmt holds data for ploting lines
